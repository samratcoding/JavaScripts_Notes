### 00. How to select DOM
```js
document.getElementById('id_name')                 // id="menu"
document.getElementsByClassName('class_name')[0]   // class="menu"
document.getElementsByTagName('html_tag_name')[0]  // h2. p, button

document.getElementBy...('parent').childNodes      // NodeList
document.getElementBy...('parent').children        // HTMLCollection

document.getElementById('myElement').matches('.className')  //  Match
document.getElementById('child').parentNode                 // parent node of selected element
document.getElementById('current').previousSibling          // previous HTML element of selected element

document.querySelector('#menu/.menu')             // with class name or id name
document.querySelector('div > p')                 //  Nested Elements:
document.querySelector('[data-role="admin"]')     // Attribute Selector, Attribute can be anything, class, id, href, name, src etc all
```

### 02. Wait for a specific DOM element
```js
// Set up a MutationObserver to watch for changes in the DOM
const observer = new MutationObserver((mutationsList, observer) => {
  const element = document.getElementById("idname");
  if (element) {
    element.click(); // Click the button when it appears
    observer.disconnect(); // Stop observing once the button is clicked
  }
});
// Start observing the document for changes
observer.observe(document.body, { childList: true, subtree: true });
```
### 03. Handle Username input in React App
```js
// Custom function to fill username in React-driven forms
function inputUsername(element, value) {
  if (element) {
    // Get the native React property for the element
    const nativeValueSetter = Object.getOwnPropertyDescriptor(element.__proto__, 'value').set;
    // Focus the input field
    element.focus();
    // Simulate React's state update by directly calling the native value setter
    nativeValueSetter.call(element, value);
    // Dispatch input events to notify React of the change
    const inputEvent = new Event('input', { bubbles: true });
    element.dispatchEvent(inputEvent);
    // Optionally trigger change event (React may use it for form submissions)
    const changeEvent = new Event('change', { bubbles: true });
    element.dispatchEvent(changeEvent);
    // Optionally blur the input field to simulate losing focus
    element.blur();
  }
}
```
```js
// uses
const username = document.getElementById('username');
inputUsername(username, request.username);
```
